#Function to change treatment levels
f <- function(x) {
  if (is.na(x) || tolower(x) %in% c("very low", "low", "moderate", 
                                    "high", "very high")) {
    return(x)
  } else {
    case_when(
      x == 0 ~ "very low",
      x == 1 ~ "low",
      x == 2 ~ "moderate",
      x == 3 ~ "high",
      x == 4 ~ "very high",
      TRUE ~ as.character(x)
    )
  }
}
me$Treatment <- sapply(me$Treatment, f)
